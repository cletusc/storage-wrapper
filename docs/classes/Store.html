Creates a substore that is nested in the current namespace.

Parameters:

  • namespace String

    The substore's namespace.

Returns:

Store:

The substore.

Example:

var Store = require('storage-wrapper');
// Create main store.
var store = new Store({
    namespace: 'myapp'
});

// Create substore.
var substore = store.createSubstore('things');
substore.set('foo', 'bar');

substore.get('foo') === store.get('things:foo');
// true

migrate

(
  • migration
)

Defined in index.js:299

Migrate data from a different namespace to current namespace.

Parameters:

  • migration Object

    The migration object.

    • toKey String

      The key name under your current namespace the old data should change to.

    • fromNamespace String

      The old namespace that the old key belongs to.

    • fromKey String

      The old key name to migrate from.

    • [fromStorageType] String optional

      The storage type to migrate from. Defaults to same type as where you are migrating to.

    • [keepOldData=false] Boolean optional

      Whether old data should be kept after it has been migrated.

    • [overwriteNewData=false] Boolean optional

      Whether old data should overwrite currently stored data if it exists.

    • [transform] Function optional

      The function to pass the old key data through before migrating.

Example:

var Store = require('storage-wrapper');
var store = new Store({
    namespace: 'myNewApp'
});

// Migrate from the old app.
store.migrate({
    toKey: 'new-key',
    fromNamespace: 'myOldApp',
    fromKey: 'old-key'
});

// Migrate from global data. Useful when moving from other storage wrappers or regular ol' localStorage.
store.migrate({
    toKey: 'other-new-key',
    fromNamespace: '',
    fromKey: 'other-old-key-on-global'
});

// Migrate some JSON data that was stored as a string.
store.migrate({
    toKey: 'new-json-key',
    fromNamespace: 'myOldApp',
    fromKey: 'old-json-key',
    // Try converting some old JSON data.
    transform: function (data) {
        try {
            return JSON.parse(data);
        }
        catch (e) {
            return data;
        }
    }
});

set

(
  • key
  • val
  • [expires]
)

Defined in index.js:177

Sets a storage item on the current namespace.

Parameters:

  • key String

    The key that the data can be accessed under.

  • val Mixed

    The value to store. May be the following types of data: RegExp, Date, Object, String, Boolean, Number

  • [expires] Date | Number optional

    The date in the future to expire, or relative number of milliseconds from Date#now to expire.

    Note: This converts special data types that normally can't be stored in the following way:

    • RegExp: prefixed with type, flags stored, and source stored as string.
    • Date: prefixed with type, stored as string using Date#toString.
    • Object: prefixed with "JSON" indicator, stored as string using JSON#stringify.
